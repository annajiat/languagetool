# List of intransitive verbs to be corrected into transitive
# الأفعال المتعدية بحرف، الخاطئة وتصحيحها التعدي بنفسها
# like استأذن منه => استأذنه
# verbs must be written with tashkeel
# نضع هنا الأفعال التي تكتب متعدية بحرف والصحيح أن تكون متعدية بنفسها
# نكتبها بالشكل:
#اِسْتَأْذَنَ=من
# الجزء الأول هو الفعل بصيغته المشكولة، تحقق من تشكيل الأصل في قاموس البرنامج، استعمل في الواجهة ctl+t كي ترى توسيم الكلمات وأصولها
# الجزء الثاني علامة المساواة
#الجزء الثالث: حرف التعدي الخاطئ، يمكن تعدد الحروف، والفصل بينها بالعمود
#الجزء الرابع: رسالة الخطأ المرفقة
اِسْتَأْذَنَ=من
اِنْتَقَصَ=من
#أَفَاضَ=في	مؤقتا للتجارب فقط
تَحَرَّى=عن	تحرّى بمعنى توخّى يتعدى مباشرة دون حرف الجر "عن"
حَجَّ=إلى	الفعل "حجّ" يتعدّى بنفسه
رَدَّ=على	الفعل "ردّ" يتعدّى بنفسه
شَكَا=من	يتعدّى الفعل "شكا" بنفسه 
طَالَعَ=في	الفعل "طالع" يتعدّى بنفسه
أَتَى=إلى	 لأن الفعل "أتى" يتعدى بنفسه مثل "أتى الحجاج إلى مكة المكرمة" والصواب "أتى الحجاج مكة المكرمة"
اِلْتَزَمَ=ب	الفعل "التزم" يتعدّى بنفسه، مثلا "التزم بنسق واضح في حديثه" والصواب "التزم نسقًا واضحًا في حديثه"
أَدَّى=ب	لأن الفعل "أدى" يتعدى بنفسه مثلا"أدت هذه الحال بفلان إلى الفقر" والصواب "أدت هذه الحال فلانًا إلى الفقر"
بَاشَرَ=في	الفعل "باشر" يتعدّى بنفسه مثلا"باشر الموظف في عمله " والصواب "باشر الموظف عمله"
تَزَوَّجَ=من	الفعل "تزوّج" يتعدّى بنفسه مثلا"تزوّج أحمدُ من حصّةَ" والصواب "تزوّج أحمدُ حصّةَ"
حَدَا=ب	الفعل "حدا" يتعدّى بنفسه مثلا"حدا به على السفر، أي "حثه وشجعه"" والصواب "حداه على السفر، أي "حثّه وشجّعه""
زَوَّدَ=ب	وذلك لأن الأصل في استعمال مثلا"زوّده بزاد وبكتاب وبشيء" والصواب "زوّده زادًا وكتابًا وشيئًا"
سَوَّلَ=ل	الفعل "سوّل" يتعدّى بنفسه  مثلا"سوّلت له نفسه بفعل هذا الأمر" والصواب "سوّلت له نفسه فعلَ هذا الأمر"
عَلَّلَ=ل	لأن الفعل "علّل" يتعدّى بنفسه دون حرف جر، ومعناه بيّن علّته، وأثبته بالدليل مثلا"علّل لما يأتي" والصواب "علّل ما يأتي"
اِعْتَقَدَ=ب	فالفعل "يعتقد" يتعدّى بنفسه من دون حرف جر مثلا"يعتقد بأن المستقبل محفوف بالأخطار" والصواب "يعتقد أن المستقبل محفوف بالأخطار"
وَافَى=ل	لأنه جاء في المعجمات اللغوية أن "وافى القوم" معناه: أتاهم، و"وافى الموت أو الكتاب فلانًا" معناه: أدركه مثلا"أرجو أن توافوني بكتابكم" والصواب "أرجو أن يوافيني كتابكم"
قَدِمَ=إلى	الخطأ "قدم زيد إلى أرض كذا"، والصواب "قَدِمَ زيدٌ أرضَ كذا"
اِسْتَفْهَمَ=من|عن	استفهم متعد بنفسه
حَازَ=على	الفعل حاز متعدٍ بنفسه
تَوَطَّنَ=في	الفعلان توطّن واستوطن متعديان بنفسيهما
اِسْتَوْطَنَ=في	الفعلان توطّن واستوطن متعديان بنفسيهما
اِخْتَصَمَ=مع	لأن ما كان على وزن "افتعل، اختصم" يقتضي المشاركة، فلا داعي لذكر "مع"، مثلا "اختصم الجار مع جاره" والصواب "اختصم الجاران"
