# List of intransitive verbs with wrong preposition to be corrected into another preposition
# الأفعال المتعدية بحرف، الخاطئة في حرف التعدية، واستبداله بحرف آخر
# like تردّد على  =>تردد إلى
# verbs must be written with tashkeel
# نضع هنا الأفعال التي تكتب متعدية بحرف والصحيح أن تكون متعدية بحرف آخر
# نكتبها بالشكل:
#تَرَدَّدَ على=إلى
# الجزء الأول هو الفعل بصيغته المشكولة مع حرف التعدية، تحقق من تشكيل الأصل في قاموس البرنامج، استعمل في الواجهة ctl+t كي ترى توسيم الكلمات وأصولها
# في حالة تعدد حرف التعدية، يضاف مدخل آخر
# الجزء الثاني علامة المساواة
#الجزء الثالث: حرف التعدي الصحيح، يمكن تعدد الحروف، والفصل بينها بالعمود
#الجزء الرابع: رسالة الخطأ المرفقة
#مثال:
#تَرَدَّدَ على=إلى	لأن الفعل "تردد" يتعدّى بـ"إلى"
تَرَدَّدَ على=إلى|في	لأن الفعل "تردد" يتعدّى بـ"إلى"
اِسْتَنَدَ على=إلى	حيث الفعل "استند" ومشتقاته يتعدّى بـ"إلى"، مثل الخطأ :"استند على كلامك، فأنت مُحقّ" والصواب "استند إلى كلامك"
تَثَبَّتَ من=في	فحرف الجر "من" ليس من الحروف الظرفية، مثل الخطأ :"تثبّت من الأمر، ينبغي التثبت من الأمر"
تَخَرَّجَ من=في	لأن تخرج في الكلية تعلم فيها وتدرب ودرس مناهجها، أما تخرج من الكلية فلا يفيد ذلك، 
خَرَجَ على=عن	الفعل "خرج" يتعدّى بحرف الجر "عن"، مثلا "خرج على القانون" والصواب "خرج عن القانون"
دَبَّ إلى=في	الفعل "دبّ" يتعدى بحرف الجر "في"، مثلا "دبّ المرض إلى جسمه" والصواب "دبّ المرض في جسمه"
دَعَا على=إلى	فالفعل "دعا" يتعدّى بحرف الجر "إلى"، مثلا "دعا زملاءه على الغداء" والصواب: "دعا زملاءه إلى الغداء"
زَادَ عن=على	الفعل "زاد" يتعدّى بـ "على"، مثلا "زاد الماء عن الزيت" والصواب " زاد الماء على الزيت"
ضَحِكَ على=من|ب	لا يتعدّى ضحك بـ"على"، مثلا "ضحك عليه بسبب أفعاله" والصواب " ضحكت منه أو به"
طَلَبَ من=إلى	لأن طلب له شيئًا وإليه كذا: سأله إياه، مثلا "طلبت من صديقي العون"، والصواب "طلبت إليه"
قَلَّ على=عن	الفعل "قل" يتعدّى بـ "عن"، مثلا "قلّ الزيت على الماء" والصواب " قلّ عن الماء"
أَذَاعَ على=بين|في	ذلك الاستخدام الصواب،مثلا " نذيع عليكم سرًّا" والصواب "نذيع بينكم أو فيكم"
وَزَّعَ على=بين|في	وَزَّعَ بين|وَزَّعَ في "أعطاهم إياهم مفرّقة"، مثلا "وزّع على التلاميذ الجوائز "أعطاهم إياهم مفرّقة"
خَفِيَ عن=على	لقوله تعالى: "إن الله لا يخفى عليه شيء" وقوله: "وما يخفى على الله من شيء"، لا يخفى عنه الخبر
اِجْتَمَعَ ب=مع	فالمصاحبة تتطلب "مع" وليس "الباء" مثلا "اجتمع المدير بالموظفين صباحًا " والصواب :"اجتمع المدير مع الموظفين صباحًا"
اِحْتَاجَ ل=إلى	يتعدّى بـ"إلى" ولا يتعدى بنفسه أو باللام مثلا "احتاج العطف والرعاية أو للعطف والرعاية" والصواب :"احتاج إلى العطف والرعاية"
أَخَذَ ب=في	 فالفعل "أخذ" يتعدى بــ "في" مثلا " أخذ بالرد على خصومه" والصواب :" أخذ في الرد على خصومه"
أَذِنَ ب=في	لأن أذن بالشيء: علمه، وأذن في الشيء: أباحه، بدليل قوله تعالى: "فأذنوا بحرب من الله ورسوله" مثلا "أذن له بالسفر" والصواب :"أذن له في السفر"
أَرْسَلَ ل=إلى	الفعل "أرسل" يتعدّى بالحرف "إلى" مثلا "أرسل له رسالة" والصواب :"أرسل إليه رسالة"
أَسِفَ ل=على	أسف على الأمر يعني "ندم على الأمر، والندم يستصحب حرف الجر "على" لا "اللام"  مثلا "أسفت للجاهل، أو أسفت للأمر المحرج" والصواب :"أسفت على الجاهل أو على الأمر المحرج "
اِشْتَاقَ ل=إلى	يتعدّى الفعل "اشتاق" بـ"إلى" مثلا "اشتقت لك بعد غيابك" والصواب :"اشتقت إليك بعد غيابك"
أَصْبَحَ على=ب	الفعل يتعدّى بحرف الجر "الباء" مثلا "أصبحنا على خير، تصبحون على خير" والصواب :"أصبحنا بخير، تصبحون بخير"
اِضْطَرَّ ل=إلى	لأن الفعل "اُضْطُرّ" يتعدّى بـ"إلى"، ويأتي مبنيًّا للمجهول مثلا "اِضْطَرّ رجل الأعمال للسفر" والصواب :"اُضْطُرّ رجل الأعمال إلى السفر"
تَنَبَّهَ إلى=ل	فالفعل يتعدّى باللام، وليس بـ "إلى" مثلا "تنبّه إلى الخطر المقبل" والصواب :"تنبّه للخطر المقبل"
خَصَّ ل=ب	فالفعل "خصّ" يتعدّى بالباء مثلا "خصصْتُه له، وخاصٌّ له" والصواب :"خصصته به ، وخاصٌّ به"
خَضَعَ إلى=ل	يتعدّى بـ"اللام" مثلا "خضع إلى تهديدات العدو" والصواب :"خضع لتهديدات العدو "
رَدَّ ل=إلى	الفعل "ردّ" يتعدّى بحرف الجر "إلى" مثلا "ردّه لمنزله" والصواب :"ردّه إلى منزله"
طَعَنَ ب=في	الفعل "طعن" يتعدّى بالحرف "في" مثلا "طعن بأقواله" والصواب :"طعن في أقواله"
فَكَّرَ ب=في	الفعل يتعدّى بـ "في" مثلا "فكّر بالعودة إلى أهله" والصواب :"فكّر في العودة إلى أهله"
نَظَرَ ل=إلى	يتعدّى "نظر" بـ"إلى" مثلا "نظر لمحمد نظرة إعجاب وإكبار " والصواب :"نظر إلى محمد نظرة إعجاب وإكبار"
نَهِمَ ل=ب	حيث يتعدّى الفعل "نَهِمَ" بالباء مثلا "نَهِمَ الرجل للمال أو إلى المال" والصواب :"نَهِمَ الرجل بالمال"
نَهِمَ إلى=ب	حيث يتعدّى الفعل "نَهِمَ" بالباء مثلا "نَهِمَ الرجل للمال أو إلى المال" والصواب :"نَهِمَ الرجل بالمال"
وَفَّقَ إلى=ل	الفعل يتعدّى بحرف الجر "اللام" مثلا "وفّقه الله إلى الخير" والصواب :"وفّقه الله للخير"
حَنَّ ل=إلى	فالفعل "حنّ" يتعدّى بحرف الجر "إلى" مثلا "يحن الإنسان لوطنه" والصواب :"يحن الإنسان إلى وطنه"
كَفَى ل=في	الفعل "يكفي" يتعدّى بـ "في" وليس بـ "اللام"  مثلا "هذا يكفي للبيان ، هو كافٍ للبيان" والصواب :"هذا يكفي في البيان، هو كافٍ في البيان"
لَاقَ ل=ب	الفعل يتعدّى بـ "الباء" مثلا "هذا الجوّال لا يليق لك" والصواب :"هذا الجوّال لا يليق بك"
